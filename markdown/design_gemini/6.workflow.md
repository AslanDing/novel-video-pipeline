# 工作流引擎与缓存系统 - 简化本地版

## 状态机设计

### WorkflowState (工作流状态)

```python
class WorkflowState(Enum):
    """工作流状态枚举"""
    IDLE = "idle"                    # 空闲
    INITIALIZING = "initializing"   # 初始化中
    RUNNING = "running"              # 运行中
    PAUSED = "paused"                # 暂停
    COMPLETED = "completed"          # 完成
    FAILED = "failed"                # 失败
    CANCELLED = "cancelled"          # 取消

class StageStatus(Enum):
    """Stage状态枚举"""
    PENDING = "pending"              # 等待中
    RUNNING = "running"              # 运行中
    COMPLETED = "completed"          # 完成
    FAILED = "failed"                # 失败
    SKIPPED = "skipped"              # 跳过
    CACHED = "cached"                # 已缓存
```

### WorkflowManager (工作流管理器)

```python
class WorkflowManager:
    """工作流管理器 - 协调四个Stage的执行"""

    def __init__(self, config: AppConfig):
        self.config = config
        self.cache_manager = CacheManager(config.cache.cache_dir)
        self.state = WorkflowState.IDLE
        self.current_stage = None
        self.stage_status = {
            1: StageStatus.PENDING,
            2: StageStatus.PENDING,
            3: StageStatus.PENDING,
            4: StageStatus.PENDING
        }
        self.logger = Logger(__name__)

        # Stage执行器
        self.stage_executors = {
            1: NovelGenerationStage(config),
            2: ImageGenerationStage(config),
            3: AudioGenerationStage(config),
            4: VideoCompositionStage(config)
        }

    def run(
        self,
        novel_name: str,
        start_stage: int = 1,
        end_stage: int = 4,
        novel_params: Dict = None
    ) -> WorkflowResult:
        """
        运行工作流

        Args:
            novel_name: 小说名称
            start_stage: 起始Stage
            end_stage: 结束Stage
            novel_params: 小说生成参数

        Returns:
            WorkflowResult: 工作流执行结果
        """
        self.state = WorkflowState.RUNNING
        start_time = time.time()

        try:
            # 检查缓存状态
            if start_stage == 1:
                # 全新开始，检查是否有完整缓存
                cached = self._check_complete_cache(novel_name)
                if cached and not self.config.force_regenerate:
                    self.logger.info(f"找到完整缓存: {novel_name}")
                    return cached

            # 执行各Stage
            context = {
                'novel_name': novel_name,
                'novel_params': novel_params or {},
                'outputs': {}
            }

            for stage_num in range(start_stage, end_stage + 1):
                self.current_stage = stage_num
                self.stage_status[stage_num] = StageStatus.RUNNING

                self.logger.info(f"开始执行 Stage {stage_num}")

                # 检查Stage缓存
                stage_cached = self._check_stage_cache(stage_num, context)
                if stage_cached and not self.config.force_regenerate:
                    self.logger.info(f"Stage {stage_num} 使用缓存")
                    self.stage_status[stage_num] = StageStatus.CACHED
                    context['outputs'][stage_num] = stage_cached
                    continue

                # 执行Stage
                executor = self.stage_executors[stage_num]
                result = executor.execute(context)

                if result.success:
                    self.stage_status[stage_num] = StageStatus.COMPLETED
                    context['outputs'][stage_num] = result.data
                    self.logger.info(f"Stage {stage_num} 完成")
                else:
                    self.stage_status[stage_num] = StageStatus.FAILED
                    raise RuntimeError(f"Stage {stage_num} 失败: {result.error}")

            # 工作流完成
            self.state = WorkflowState.COMPLETED
            elapsed_time = time.time() - start_time

            return WorkflowResult(
                success=True,
                novel_name=novel_name,
                elapsed_time=elapsed_time,
                stage_status=self.stage_status,
                outputs=context['outputs']
            )

        except Exception as e:
            self.state = WorkflowState.FAILED
            self.logger.error(f"工作流失败: {e}")

            return WorkflowResult(
                success=False,
                novel_name=novel_name,
                error=str(e),
                stage_status=self.stage_status
            )

    def pause(self) -> None:
        """暂停工作流"""
        if self.state == WorkflowState.RUNNING:
            self.state = WorkflowState.PAUSED
            self.logger.info("工作流已暂停")

    def resume(self) -> None:
        """恢复工作流"""
        if self.state == WorkflowState.PAUSED:
            self.state = WorkflowState.RUNNING
            self.logger.info("工作流已恢复")

    def cancel(self) -> None:
        """取消工作流"""
        self.state = WorkflowState.CANCELLED
        self.logger.info("工作流已取消")

    def get_status(self) -> WorkflowStatus:
        """获取当前状态"""
        return WorkflowStatus(
            state=self.state,
            current_stage=self.current_stage,
            stage_status=self.stage_status,
            progress=self._calculate_progress()
        )

    def _calculate_progress(self) -> float:
        """计算整体进度"""
        if not self.current_stage:
            return 0.0

        stage_weights = {1: 0.4, 2: 0.3, 3: 0.2, 4: 0.1}

        completed_stages = sum(
            1 for status in self.stage_status.values()
            if status in (StageStatus.COMPLETED, StageStatus.CACHED)
        )

        current_stage_progress = 0.0
        if self.current_stage and self.stage_status[self.current_stage] == StageStatus.RUNNING:
            current_stage_progress = 0.5  # 假设当前Stage完成50%

        total_progress = (
            sum(stage_weights[i] for i in range(1, completed_stages + 1)) +
            stage_weights.get(self.current_stage, 0) * current_stage_progress
        )

        return min(total_progress, 1.0)
```

## 缓存系统设计

### CacheManager (缓存管理器)

```python
class CacheManager:
    """缓存管理器 - 统一管理各级缓存"""

    def __init__(self, cache_dir: str):
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(parents=True, exist_ok=True)
        self.memory_cache = {}
        self.max_memory_size = 100 * 1024 * 1024  # 100MB
        self.current_memory_size = 0

    def get(self, key: str, level: str = "all") -> Optional[Any]:
        """
        获取缓存

        Args:
            key: 缓存键
            level: 缓存级别 ("memory", "disk", "all")
        """
        # L1: 内存缓存
        if level in ("memory", "all"):
            if key in self.memory_cache:
                return self.memory_cache[key]['data']

        # L2: 磁盘缓存
        if level in ("disk", "all"):
            cache_path = self._get_disk_cache_path(key)
            if cache_path.exists():
                try:
                    with open(cache_path, 'r', encoding='utf-8') as f:
                        data = json.load(f)

                    # 同时存入内存缓存
                    self._set_memory_cache(key, data)

                    return data
                except Exception as e:
                    logger.warning(f"读取缓存失败 {key}: {e}")

        return None

    def set(
        self,
        key: str,
        data: Any,
        level: str = "all",
        metadata: Dict = None
    ) -> None:
        """
        设置缓存

        Args:
            key: 缓存键
            data: 缓存数据
            level: 缓存级别 ("memory", "disk", "all")
            metadata: 额外元数据
        """
        # L1: 内存缓存
        if level in ("memory", "all"):
            self._set_memory_cache(key, data, metadata)

        # L2: 磁盘缓存
        if level in ("disk", "all"):
            self._set_disk_cache(key, data, metadata)

    def invalidate(self, pattern: str) -> int:
        """
        失效匹配pattern的缓存

        Returns:
            失效的缓存数量
        """
        count = 0

        # 清理内存缓存
        keys_to_remove = [k for k in self.memory_cache if pattern in k]
        for k in keys_to_remove:
            del self.memory_cache[k]
            count += 1

        # 清理磁盘缓存
        for cache_file in self.cache_dir.rglob("*.json"):
            if pattern in cache_file.name:
                cache_file.unlink()
                count += 1

        return count

    def clear_all(self) -> int:
        """清空所有缓存"""
        # 清空内存缓存
        count = len(self.memory_cache)
        self.memory_cache.clear()
        self.current_memory_size = 0

        # 清空磁盘缓存
        for cache_file in self.cache_dir.rglob("*"):
            if cache_file.is_file():
                cache_file.unlink()
                count += 1

        return count

    def get_stats(self) -> Dict:
        """获取缓存统计"""
        disk_size = sum(
            f.stat().st_size
            for f in self.cache_dir.rglob("*")
            if f.is_file()
        )

        return {
            "memory_entries": len(self.memory_cache),
            "memory_size_mb": self.current_memory_size / (1024 * 1024),
            "disk_entries": len(list(self.cache_dir.rglob("*.json"))),
            "disk_size_mb": disk_size / (1024 * 1024),
        }

    # 内部方法

    def _set_memory_cache(
        self,
        key: str,
        data: Any,
        metadata: Dict = None
    ) -> None:
        """设置内存缓存"""
        # 估算数据大小
        data_size = len(json.dumps(data, ensure_ascii=False).encode('utf-8'))

        # 检查是否超出限制
        if self.current_memory_size + data_size > self.max_memory_size:
            self._evict_memory_cache(data_size)

        # 存入缓存
        self.memory_cache[key] = {
            'data': data,
            'metadata': metadata or {},
            'size': data_size,
            'timestamp': time.time()
        }
        self.current_memory_size += data_size

    def _set_disk_cache(
        self,
        key: str,
        data: Any,
        metadata: Dict = None
    ) -> None:
        """设置磁盘缓存"""
        cache_path = self._get_disk_cache_path(key)
        cache_path.parent.mkdir(parents=True, exist_ok=True)

        cache_data = {
            'data': data,
            'metadata': metadata or {},
            'created_at': time.time(),
            'version': '1.0'
        }

        with open(cache_path, 'w', encoding='utf-8') as f:
            json.dump(cache_data, f, ensure_ascii=False, indent=2)

    def _evict_memory_cache(self, required_size: int) -> None:
        """淘汰内存缓存"""
        # 按时间戳排序，淘汰最旧的
        sorted_items = sorted(
            self.memory_cache.items(),
            key=lambda x: x[1]['timestamp']
        )

        freed_size = 0
        for key, item in sorted_items:
            if freed_size >= required_size:
                break

            del self.memory_cache[key]
            freed_size += item['size']
            self.current_memory_size -= item['size']

    def _get_disk_cache_path(self, key: str) -> Path:
        """获取磁盘缓存路径"""
        # 使用哈希分布避免单目录文件过多
        hash_prefix = hashlib.md5(key.encode()).hexdigest()[:2]
        return self.cache_dir / hash_prefix / f"{key}.json"
```

## 主入口设计

### main.py (统一入口)

```python
#!/usr/bin/env python3
"""
AI小说视频生成系统 - 主入口

统一命令行入口，支持完整的四阶段流程。
"""

import sys
import argparse
from pathlib import Path

from config.settings import load_config
from core.workflow import WorkflowManager
from core.logger import setup_logger


def main():
    parser = argparse.ArgumentParser(
        description="AI小说视频生成系统",
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog="""
示例:
  # 完整流程
  python main.py --novel "我的修仙小说" --genre 修仙 --chapters 3

  # 只生成小说
  python main.py --novel "我的修仙小说" --stage 1

  # 从第二阶段开始
  python main.py --novel "我的修仙小说" --from-stage 2

  # 强制重新生成
  python main.py --novel "我的修仙小说" --force
        """
    )

    # 基本参数
    parser.add_argument('-n', '--novel', required=True, help='小说名称')
    parser.add_argument('-g', '--genre', default='修仙', help='小说类型')
    parser.add_argument('-c', '--chapters', type=int, default=3, help='章节数')
    parser.add_argument('-s', '--scenes-per-chapter', type=int, default=3,
                        help='每章场景数')

    # Stage控制
    stage_group = parser.add_mutually_exclusive_group()
    stage_group.add_argument('--stage', type=int, choices=[1,2,3,4],
                            help='只执行指定Stage')
    stage_group.add_argument('--from-stage', type=int, choices=[1,2,3,4],
                            help='从指定Stage开始')
    stage_group.add_argument('--to-stage', type=int, choices=[1,2,3,4],
                            help='执行到指定Stage')

    # 缓存和调试
    parser.add_argument('--force', action='store_true',
                       help='强制重新生成，忽略缓存')
    parser.add_argument('--clear-cache', action='store_true',
                       help='清空所有缓存')
    parser.add_argument('--cache-dir', type=str,
                       help='指定缓存目录')
    parser.add_argument('-v', '--verbose', action='store_true',
                       help='显示详细日志')
    parser.add_argument('--debug', action='store_true',
                       help='启用调试模式')

    # 视频参数
    parser.add_argument('--video-mode', choices=['compose', 'svd'], default='compose',
                       help='视频生成模式')
    parser.add_argument('--no-subtitles', action='store_true',
                       help='不生成字幕')

    args = parser.parse_args()

    # 设置日志
    log_level = 'DEBUG' if args.debug else ('INFO' if args.verbose else 'WARNING')
    setup_logger(log_level)

    # 加载配置
    config = load_config()

    # 应用命令行参数
    if args.cache_dir:
        config.cache.cache_dir = args.cache_dir
    if args.force:
        config.force_regenerate = True

    # 清空缓存（如果需要）
    if args.clear_cache:
        cache_manager = CacheManager(config.cache.cache_dir)
        count = cache_manager.clear_all()
        print(f"已清空 {count} 个缓存项")
        return 0

    # 确定Stage范围
    start_stage = args.from_stage or args.stage or 1
    end_stage = args.to_stage or args.stage or 4

    if args.stage:
        start_stage = end_stage = args.stage

    # 构建小说参数
    novel_params = {
        'genre': args.genre,
        'chapters': args.chapters,
        'scenes_per_chapter': args.scenes_per_chapter
    }

    # 视频参数
    video_params = {
        'mode': args.video_mode,
        'subtitles': not args.no_subtitles
    }

    # 运行工作流
    try:
        workflow = WorkflowManager(config)

        result = workflow.run(
            novel_name=args.novel,
            start_stage=start_stage,
            end_stage=end_stage,
            novel_params=novel_params,
            video_params=video_params
        )

        if result.success:
            print(f"\n✅ 工作流完成!")
            print(f"   小说: {result.novel_name}")
            print(f"   耗时: {result.elapsed_time:.2f}秒")
            print(f"   输出目录: outputs/")
            return 0
        else:
            print(f"\n❌ 工作流失败: {result.error}")
            return 1

    except KeyboardInterrupt:
        print("\n⚠️ 用户中断")
        workflow.cancel() if 'workflow' in locals() else None
        return 130

    except Exception as e:
        print(f"\n❌ 错误: {e}")
        if args.debug:
            import traceback
            traceback.print_exc()
        return 1


if __name__ == '__main__':
    sys.exit(main())
```
