# Stage 3: 音频生成引擎 - 简化本地版

## 概述

从小说JSON提取对话和旁白文本，使用TTS引擎生成音频，支持角色语音映射。

## 技术选型配置

| 配置项 | 默认值 | 说明 |
|--------|--------|------|
| **主TTS引擎** | Fish Audio S2 Pro | 高质量多角色配音 |
| **备选TTS** | CosyVoice 3.0 | 中文旁白优化 |
| **保底方案** | Edge-TTS | 无需显存，多音色 |
| **采样率** | 44100 Hz | CD音质 |
| **格式** | WAV / MP3 | 无损压缩兼顾 |

## 输入输出

### 输入
- Stage 1 生成的小说JSON: `outputs/novels/{name}/novel.json`
- 可选: 角色语音配置文件

### 输出
- 分段音频文件: `outputs/audio/{chapter}/segment_{n}.mp3`
- 合并后的章节音频: `outputs/audio/{chapter}/combined.mp3`
- 音频元数据: `outputs/audio/{chapter}/metadata.json`

## 文本清理

### 清理规则

```python
class TextCleaner:
    """文本清理器 - 移除TTS不友好的字符"""

    # 需要移除的模式
    REMOVE_PATTERNS = [
        r'#+\s*',                    # Markdown标题
        r'\*+',                      # 粗体/斜体标记
        r'`[^`]*`',                  # 行内代码
        r'!?\[([^\]]*)\]\([^)]+\)',  # 图片/链接
        r'<[^>]+>',                  # HTML标签
        r'https?://\S+',             # URL
        r'[@#]\w+',                  # @提及 和 #话题
        r'[\U0001F600-\U0001F64F\U0001F300-\U0001F5FF\U0001F680-\U0001F6FF\U0001F1E0-\U0001F1FF]',  # emoji
    ]

    # 替换规则
    REPLACE_PATTERNS = {
        r'\n+': ' ',           # 多个换行 -> 空格
        r'\s+': ' ',          # 多个空格 -> 单个空格
        r'[.]{3,}': '…',      # 多个点 -> 省略号
        r'[!]{2,}': '!',      # 多个感叹号 -> 单个
        r'[?]{2,}': '?',      # 多个问号 -> 单个
    }

    @classmethod
    def clean(cls, text: str) -> str:
        """清理文本"""
        if not text:
            return ""

        # 应用移除规则
        for pattern in cls.REMOVE_PATTERNS:
            text = re.sub(pattern, '', text, flags=re.MULTILINE)

        # 应用替换规则
        for pattern, replacement in cls.REPLACE_PATTERNS.items():
            text = re.sub(pattern, replacement, text)

        # 最终清理
        text = text.strip()

        # 限制长度（TTS通常有长度限制）
        if len(text) > 5000:
            text = text[:5000]

        return text
```

## 核心组件

### 1. AudioGenerator (音频生成器)

```python
class AudioGenerator:
    """音频生成主类"""

    def __init__(self, config: AudioGenConfig):
        self.config = config
        self.cache_manager = CacheManager(config.cache_dir)
        self.text_cleaner = TextCleaner()
        self.voice_mapper = VoiceMapper(config.voice_mapping)
        self.tts_engine = self._init_tts_engine()

    def generate(self, novel_json: Path, output_dir: Path) -> AudioOutput:
        """
        完整音频生成流程

        Args:
            novel_json: 小说JSON文件路径
            output_dir: 输出目录

        Returns:
            AudioOutput: 生成结果
        """
        # 1. 加载小说数据
        novel_data = self._load_novel(novel_json)

        # 2. 提取音频分段
        segments = self._extract_segments(novel_data)

        # 3. 按章节生成
        results = []
        for chapter_num in range(1, novel_data['metadata']['total_chapters'] + 1):
            chapter_segments = [s for s in segments if s.chapter_number == chapter_num]

            chapter_results = self._generate_chapter_audio(
                chapter_num=chapter_num,
                segments=chapter_segments,
                output_dir=output_dir / f"chapter_{chapter_num:03d}"
            )
            results.extend(chapter_results)

        # 4. 保存元数据
        self._save_metadata(results, output_dir)

        return AudioOutput(
            total_segments=len(segments),
            generated_segments=len([r for r in results if r.success]),
            total_duration=sum([r.duration for r in results if r.duration]),
            output_dir=output_dir
        )

    def _extract_segments(self, novel_data: Dict) -> List[AudioSegment]:
        """从小说数据提取音频分段"""
        segments = []

        for chapter in novel_data['chapters']:
            chapter_num = chapter['chapter_number']

            # 使用预生成的audio_segments
            for i, seg_data in enumerate(chapter.get('audio_segments', [])):
                segment = AudioSegment(
                    segment_id=f"ch{chapter_num}_seg{i+1}",
                    chapter_number=chapter_num,
                    segment_number=i+1,
                    text=seg_data['text'],
                    cleaned_text=self.text_cleaner.clean(seg_data['text']),
                    speaker=seg_data.get('speaker', '旁白'),
                    voice_id=self.voice_mapper.get_voice_id(seg_data.get('speaker', '旁白')),
                    emotion=seg_data.get('emotion', 'neutral')
                )
                segments.append(segment)

        return segments

    def _generate_chapter_audio(
        self,
        chapter_num: int,
        segments: List[AudioSegment],
        output_dir: Path
    ) -> List[GenerationResult]:
        """生成单个章节的所有音频"""
        output_dir.mkdir(parents=True, exist_ok=True)
        results = []
        audio_files = []

        for segment in tqdm(segments, desc=f"Chapter {chapter_num} Audio"):
            # 检查缓存
            cached = self._check_audio_cache(segment)
            if cached:
                results.append(cached)
                if cached.audio_path:
                    audio_files.append(cached.audio_path)
                continue

            # 生成音频
            try:
                audio_path = output_dir / f"{segment.segment_id}.mp3"

                self.tts_engine.generate(
                    text=segment.cleaned_text,
                    voice_id=segment.voice_id,
                    output_path=audio_path,
                    speed=self.config.speed,
                    volume=self.config.volume
                )

                # 获取音频时长
                duration = self._get_audio_duration(audio_path)

                result = GenerationResult(
                    segment_id=segment.segment_id,
                    success=True,
                    audio_path=str(audio_path),
                    duration=duration
                )

                audio_files.append(str(audio_path))

            except Exception as e:
                result = GenerationResult(
                    segment_id=segment.segment_id,
                    success=False,
                    error=str(e)
                )

            # 保存缓存
            self._save_audio_cache(segment, result)
            results.append(result)

        # 合并章节音频
        if len(audio_files) > 1:
            combined_path = output_dir / "combined.mp3"
            self._merge_audio_files(audio_files, combined_path)

        return results

    def _merge_audio_files(self, audio_files: List[str], output_path: Path) -> None:
        """合并多个音频文件"""
        # 使用pydub或ffmpeg合并
        from pydub import AudioSegment

        combined = AudioSegment.empty()
        for audio_file in audio_files:
            segment = AudioSegment.from_mp3(audio_file)
            combined += segment

        combined.export(output_path, format="mp3")
```

### 2. VoiceMapper (声音映射器)

```python
class VoiceMapper:
    """角色到声音ID的映射管理"""

    def __init__(self, mapping_config: Dict = None):
        self.default_voice = "zh-CN-XiaoxiaoNeural"

        # 默认角色-声音映射
        self.role_voice_map = {
            "旁白": "zh-CN-YunxiNeural",
            "主角": "zh-CN-YunxiNeural",
            "女主角": "zh-CN-XiaoxiaoNeural",
            "老者": "zh-CN-YunjianNeural",
            "少年": "zh-CN-YunxiNeural"
        }

        # 合并自定义映射
        if mapping_config:
            self.role_voice_map.update(mapping_config)

    def get_voice_id(self, speaker: str) -> str:
        """获取角色对应的声音ID"""
        return self.role_voice_map.get(speaker, self.default_voice)

    def register_voice(self, role: str, voice_id: str) -> None:
        """注册角色声音映射"""
        self.role_voice_map[role] = voice_id

    def list_available_voices(self) -> Dict[str, List[str]]:
        """列出可用声音"""
        return {
            "edge_tts_zh": [
```

### 3. TTSEngine (TTS引擎)

```python
class TTSEngine:
    """TTS引擎抽象层"""

    def __init__(self, config: AudioGenConfig):
        self.config = config
        self.provider = config.provider
        self._engine = None

    def _get_engine(self):
        """延迟加载TTS引擎"""
        if self._engine is None:
            if self.provider == "edge-tts":
                self._engine = EdgeTTSEngine()
            elif self.provider == "chattts":
                self._engine = ChatTTSEngine()
            else:
                raise ValueError(f"Unknown TTS provider: {self.provider}")
        return self._engine

    def generate(
        self,
        text: str,
        voice_id: str,
        output_path: Path,
        speed: float = 1.0,
        volume: float = 1.0
    ) -> None:
        """生成音频"""
        engine = self._get_engine()
        engine.generate(
            text=text,
            voice_id=voice_id,
            output_path=output_path,
            speed=speed,
            volume=volume
        )

class EdgeTTSEngine:
    """Edge TTS引擎实现"""

    def generate(
        self,
        text: str,
        voice_id: str,
        output_path: Path,
        speed: float = 1.0,
        volume: float = 1.0
    ) -> None:
        """使用edge-tts生成音频"""
        import edge_tts
        import asyncio

        async def _generate():
            communicate = edge_tts.Communicate(
                text=text,
                voice=voice_id,
                rate=f"{int((speed-1)*100)}%",
                volume=f"{int((volume-1)*100)}%"
            )
            await communicate.save(str(output_path))

        asyncio.run(_generate())
```

## 命令行接口

```bash

python run_stage3.py --novel-dir "outputs/novels/我的修仙小说"

python run_stage3.py --novel-dir "outputs/novels/我的修仙小说" --chapter 1

python run_stage3.py --novel-dir "outputs/novels/我的修仙小说" --force

python run_stage3.py --novel-dir "outputs/novels/我的修仙小说" --tts edge-tts
```
