```python
# image_generator.py:595-636
async def _load_scenes_from_script(self, novel_title: str, chapter_number: int) -> List[Dict]:
    """从 Stage 1 产出的 script_x.jsonl 加载场景"""
    from stages.stage2_visual.script_adapter import ScriptAdapter

    adapter = ScriptAdapter(novel_title, data_dir)
    script_lines = adapter.load_script_lines(chapter_number)
    prompts = adapter.extract_visual_prompts(script_lines)

    # 转换为场景格式
    scenes = []
    for i, p in enumerate(prompts):
        scenes.append({
            'visual_prompt': p.get('visual_prompt', ''),
            'shot_id': p.get('shot_id', ''),
            'scene_id': p.get('scene_id', ''),
            # ...
        })
    return scenes
```
