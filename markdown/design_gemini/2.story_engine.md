# Stage 1: 小说生成引擎 - 简化本地版

## 概述

基于LLM的小说生成系统，使用结构化输出确保可控性，支持完整的缓存机制。

## 技术选型配置

| 配置项 | 默认值 | 说明 |
|--------|--------|------|
| **模型** | Qwen3-14B-Instruct-AWQ | 14B量级，适合24GB单卡 |
| **推理框架** | vLLM | 高效推理，OpenAI兼容API |
| **上下文长度** | 32K | 支持长上下文小说生成 |
| **量化** | AWQ INT4 | 降低显存占用 |
| **温度** | 0.7 | 创造性vs确定性平衡 |

**备选模型**: Qwen2.5-14B-Instruct、Yi-1.5-9B-Chat

## 关键类设计

### NovelGenerator

```python
class NovelGenerator:
    """小说生成器主类"""

    def __init__(self, config: NovelConfig):
        self.config = config
        self.cache_manager = CacheManager(config.cache_dir)
        self.llm_client = LLMClient(config.llm_config)
        self.logger = Logger(__name__)

    def generate(self, prompt: str) -> Novel:
        """完整生成流程"""
        # 1. 检查完整缓存
        if self._check_complete_cache():
            return self._load_from_cache()

        # 2. 逐步生成
        worldbuilding = self._generate_worldbuilding()
        characters = self._generate_characters(worldbuilding)
        plot = self._generate_plot(worldbuilding, characters)

        # 3. 章节生成（支持断点续写）
        chapters = []
        for i in range(self.config.num_chapters):
            chapter = self._generate_chapter(
                chapter_num=i+1,
                worldbuilding=worldbuilding,
                characters=characters,
                plot=plot
            )
            chapters.append(chapter)

        # 4. 组装并保存
        novel = Novel(
            metadata=self._create_metadata(),
            worldbuilding=worldbuilding,
            characters=characters,
            plot=plot,
            chapters=chapters
        )

        self._save_complete_novel(novel)
        return novel
```

### CacheManager

```python
class CacheManager:
    """缓存管理器"""

    def __init__(self, cache_dir: str):
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(parents=True, exist_ok=True)
        self.memory_cache = {}

    def get(self, key: str, level: str = "disk") -> Optional[Any]:
        """获取缓存"""
        # L1: 内存缓存
        if key in self.memory_cache:
            return self.memory_cache[key]

        if level == "memory":
            return None

        # L2: 磁盘缓存
        cache_path = self._get_cache_path(key)
        if cache_path.exists():
            with open(cache_path, 'r', encoding='utf-8') as f:
                data = json.load(f)
                # 存入内存缓存
                self.memory_cache[key] = data
                return data

        return None

    def set(self, key: str, value: Any, level: str = "both") -> None:
        """设置缓存"""
        # L1: 内存缓存
        if level in ("memory", "both"):
            self.memory_cache[key] = value

        # L2: 磁盘缓存
        if level in ("disk", "both"):
            cache_path = self._get_cache_path(key)
            cache_path.parent.mkdir(parents=True, exist_ok=True)
            with open(cache_path, 'w', encoding='utf-8') as f:
                json.dump(value, f, ensure_ascii=False, indent=2)

    def invalidate(self, pattern: str) -> None:
        """失效缓存"""
        # 清除匹配的内存缓存
        keys_to_remove = [k for k in self.memory_cache if pattern in k]
        for k in keys_to_remove:
            del self.memory_cache[k]

        # 清除匹配的磁盘缓存
        for cache_file in self.cache_dir.rglob("*.json"):
            if pattern in cache_file.name:
                cache_file.unlink()
```

## 配置文件示例

```python
# config/settings.py

from pydantic import BaseModel
from typing import Optional, List

class LLMConfig(BaseModel):
    """LLM配置"""
    provider: str = "local"  # local, openai, ollama
    model_name: str = "Qwen2.5-7B-Instruct"
    api_base: Optional[str] = "http://localhost:11434"
    api_key: Optional[str] = None
    temperature: float = 0.7
    max_tokens: int = 4000

class ImageGenConfig(BaseModel):
    """图像生成配置"""
    enabled: bool = True
    model: str = "sdxl"  # sdxl, flux, z-image-turbo
    local_model_path: str = "./models/sdxl"
    use_ipadapter: bool = True
    ipadapter_model: str = "./models/ipadapter"
    default_width: int = 1024
    default_height: int = 1024
    steps: int = 30
    cfg_scale: float = 7.0

class AudioGenConfig(BaseModel):
    """音频生成配置"""
    enabled: bool = True
    provider: str = "edge-tts"  # edge-tts, chattts
    default_voice: str = "zh-CN-XiaoxiaoNeural"
    speed: float = 1.0
    volume: float = 1.0

class VideoGenConfig(BaseModel):
    """视频生成配置"""
    enabled: bool = True
    mode: str = "compose"  # compose, svd
    ffmpeg_path: str = "ffmpeg"
    fps: int = 24
    resolution: str = "1080p"  # 720p, 1080p
    transition_duration: float = 0.5

class CacheConfig(BaseModel):
    """缓存配置"""
    enabled: bool = True
    cache_dir: str = "./cache"
    max_memory_cache_size: int = 100  # MB
    ttl: int = 86400  # 24小时

class AppConfig(BaseModel):
    """应用主配置"""
    debug: bool = False
    log_level: str = "INFO"
    output_dir: str = "./outputs"
    max_workers: int = 4

    llm: LLMConfig = LLMConfig()
    image: ImageGenConfig = ImageGenConfig()
    audio: AudioGenConfig = AudioGenConfig()
    video: VideoGenConfig = VideoGenConfig()
    cache: CacheConfig = CacheConfig()
    app: AppConfig = AppConfig()
```

## 总结

本设计文档描述了一个简化的本地小说视频生成系统，核心特点：

1. **无Web界面**: 纯命令行操作
2. **完整缓存**: 支持断点续传和调试
3. **本地模型**: 除LLM外全部本地运行
4. **简化架构**: 去除不必要的微服务和复杂架构
5. **便于调试**: 详细的日志和中间产物
