# Stage 2: 图像生成引擎 - 简化本地版

## 概述

从小说JSON提取场景描述，使用本地SDXL/Flux模型生成图像，支持角色一致性控制。

## 技术选型配置

| 配置项 | 默认值 | 说明 |
|--------|--------|------|
| **基础模型** | Stable Diffusion 3.5 Medium | 高质量图像生成 |
| **推理框架** | diffusers | HuggingFace官方框架 |
| **分辨率** | 1024x1024 | SD3.5原生分辨率 |
| **步数** | 30-50 | 质量vs速度平衡 |
| **CFG Scale** | 7.0 | 提示词遵循度 |

**角色一致性控制**:

| 技术 | 用途 | 配置 |
|------|------|------|
| **IP-Adapter** | 角色面部一致性 | 权重: 0.6-0.8 |
| **ControlNet** | 姿态/构图控制 | OpenPose + Canny |
| **Reference-only** | 风格统一 | 启用 attention耦合 |

**备选模型**: SDXL 1.0、Flux.1-schnell

## 输入输出

### 输入
- Stage 1 生成的小说JSON文件: `outputs/novels/{name}/novel.json`
- 角色参考图像 (可选): `outputs/novels/{name}/characters/{char_id}.png`

### 输出
- 生成的场景图像: `outputs/images/{chapter}/{scene_id}.png`
- 图像生成元数据: `outputs/images/{chapter}/metadata.json`

## 核心组件

### 1. ImageGenerator (图像生成器)

```python
class ImageGenerator:
    """图像生成主类"""

    def __init__(self, config: ImageGenConfig):
        self.config = config
        self.cache_manager = CacheManager(config.cache_dir)
        self.prompt_optimizer = PromptOptimizer()
        self.character_consistency = CharacterConsistencyManager()
        self.model_pipeline = None  # 延迟加载

    def generate(self, novel_json: Path, output_dir: Path) -> ImageOutput:
        """
        完整图像生成流程

        Args:
            novel_json: 小说JSON文件路径
            output_dir: 输出目录

        Returns:
            ImageOutput: 生成结果
        """
        # 1. 加载小说数据
        novel_data = self._load_novel(novel_json)

        # 2. 提取所有场景
        scenes = self._extract_scenes(novel_data)

        # 3. 预加载角色一致性参考
        character_refs = self._load_character_references(novel_data)

        # 4. 按章节批量生成
        results = []
        for chapter_num in range(1, novel_data['metadata']['total_chapters'] + 1):
            chapter_scenes = [s for s in scenes if s.chapter_number == chapter_num]

            chapter_results = self._generate_chapter_images(
                chapter_num=chapter_num,
                scenes=chapter_scenes,
                character_refs=character_refs,
                output_dir=output_dir / f"chapter_{chapter_num:03d}"
            )
            results.extend(chapter_results)

        # 5. 保存元数据
        self._save_metadata(results, output_dir)

        return ImageOutput(
            total_scenes=len(scenes),
            generated_scenes=len([r for r in results if r.success]),
            failed_scenes=[r.scene_id for r in results if not r.success],
            output_dir=output_dir
        )

    def _generate_chapter_images(
        self,
        chapter_num: int,
        scenes: List[SceneImage],
        character_refs: Dict[str, Path],
        output_dir: Path
    ) -> List[GenerationResult]:
        """生成单个章节的所有图像"""
        output_dir.mkdir(parents=True, exist_ok=True)
        results = []

        for scene in tqdm(scenes, desc=f"Chapter {chapter_num}"):
            # 检查缓存
            cached = self._check_cache(scene)
            if cached:
                results.append(cached)
                continue

            # 优化prompt
            optimized_prompt = self.prompt_optimizer.optimize(
                description=scene.source_description,
                characters=scene.character_ids,
                mood=scene.mood
            )

            # 生成图像
            try:
                image_path = self._generate_single_image(
                    prompt=optimized_prompt,
                    character_refs={cid: character_refs.get(cid)
                                   for cid in scene.character_ids},
                    output_path=output_dir / f"{scene.scene_id}.png"
                )

                result = GenerationResult(
                    scene_id=scene.scene_id,
                    success=True,
                    image_path=image_path,
                    prompt=optimized_prompt
                )

            except Exception as e:
                result = GenerationResult(
                    scene_id=scene.scene_id,
                    success=False,
                    error=str(e)
                )

            # 保存缓存
            self._save_cache(scene, result)
            results.append(result)

        return results

    def _generate_single_image(
        self,
        prompt: str,
        character_refs: Dict[str, Path],
        output_path: Path,
        negative_prompt: str = "",
        seed: int = None
    ) -> Path:
        """生成单张图像"""
        if self.model_pipeline is None:
            self._load_model()

        # 设置随机种子
        if seed is None:
            seed = random.randint(0, 2**32)

        generator = torch.Generator(self.config.device).manual_seed(seed)

        # 构建生成参数
        generation_args = {
            "prompt": prompt,
            "negative_prompt": negative_prompt or self.config.default_negative_prompt,
            "num_inference_steps": self.config.steps,
            "guidance_scale": self.config.cfg_scale,
            "generator": generator,
            "width": self.config.default_width,
            "height": self.config.default_height
        }

        # 应用角色一致性 (IPAdapter)
        if character_refs and self.config.use_ipadapter:
            generation_args = self._apply_ipadapter(
                generation_args, character_refs
            )

        # 生成图像
        image = self.model_pipeline(**generation_args).images[0]

        # 保存
        output_path.parent.mkdir(parents=True, exist_ok=True)
        image.save(output_path, quality=95)

        return output_path
```

### 2. PromptOptimizer (Prompt优化器)

```python
class PromptOptimizer:
    """场景描述到图像Prompt的优化器"""

    def __init__(self):
        self.style_presets = {
            "xianxia": {
                "prefix": "masterpiece, best quality, xianxia fantasy art, chinese cultivation world,",
                "suffix": "ethereal atmosphere, glowing spiritual energy, mystical clouds, ancient architecture, dramatic lighting"
            },
            "fantasy": {
                "prefix": "masterpiece, best quality, high fantasy art,",
                "suffix": "dramatic lighting, highly detailed, cinematic composition"
            },
            "realistic": {
                "prefix": "masterpiece, best quality, photorealistic,",
                "suffix": "professional photography, 8k uhd, dslr, soft lighting, high quality"
            }
        }

        self.character_prompts = {
            "protagonist": "dynamic pose, determined expression, heroic aura",
            "beauty": "elegant, flowing hair, ethereal beauty, refined features",
            "elder": "wise expression, white hair, ancient aura, profound eyes"
        }

    def optimize(
        self,
        description: str,
        characters: List[str] = None,
        mood: str = "neutral",
        style: str = "xianxia"
    ) -> str:
        """
        优化场景描述为图像生成prompt

        Args:
            description: 原始场景描述
            characters: 出现的角色ID列表
            mood: 场景氛围
            style: 风格预设

        Returns:
            优化后的prompt
        """
        preset = self.style_presets.get(style, self.style_presets["xianxia"])

        # 构建角色描述
        char_prompts = []
        if characters:
            for char_id in characters:
                char_desc = self._get_character_description(char_id)
                if char_desc:
                    char_prompts.append(char_desc)

        # 构建氛围描述
        mood_prompt = self._get_mood_prompt(mood)

        # 组装最终prompt
        parts = [
            preset["prefix"],
            description,
            ",".join(char_prompts) if char_prompts else "",
            mood_prompt,
            preset["suffix"]
        ]

        # 清理和格式化
        prompt = ", ".join([p for p in parts if p])
        prompt = self._clean_prompt(prompt)

        return prompt

    def _get_character_description(self, char_id: str) -> str:
        """获取角色描述"""
        # 从角色缓存中获取信息
        # 简化版本，实际应从角色管理器获取
        return self.character_prompts.get(char_id, "")

    def _get_mood_prompt(self, mood: str) -> str:
        """获取氛围描述"""
        mood_map = {
            "tense": "tense atmosphere, dramatic lighting, suspenseful",
            "peaceful": "peaceful atmosphere, soft lighting, serene",
            "epic": "epic scale, grand atmosphere, majestic lighting",
            "romantic": "romantic atmosphere, warm lighting, intimate",
            "mysterious": "mysterious atmosphere, dark lighting, enigmatic"
        }
        return mood_map.get(mood, "")

    def _clean_prompt(self, prompt: str) -> str:
        """清理prompt"""
        # 移除多余空格
        prompt = " ".join(prompt.split())
        # 修复逗号
        prompt = prompt.replace(" ,", ",").replace(",,", ",")
        return prompt.strip()
```

### 3. CharacterConsistencyManager (角色一致性管理器)

```python
class CharacterConsistencyManager:
    """管理角色一致性，使用IPAdapter或Regional Prompting"""

    def __init__(self, config: ImageGenConfig):
        self.config = config
        self.reference_images = {}  # char_id -> image_path
        self.ipadapter_pipeline = None

    def load_character_reference(
        self,
        char_id: str,
        image_path: Path
    ) -> None:
        """加载角色参考图像"""
        if not image_path.exists():
            raise FileNotFoundError(f"参考图像不存在: {image_path}")

        self.reference_images[char_id] = image_path

    def generate_character_reference(
        self,
        char_info: Character,
        output_path: Path
    ) -> Path:
        """
        从角色描述生成参考图像

        当没有提供角色参考图时使用，
        基于角色描述生成一张标准参考图
        """
        prompt = self._build_character_prompt(char_info)

        # 使用基础模型生成
        image = self._generate_base_image(
            prompt=prompt,
            size=(512, 768)  # 人物 portrait 尺寸
        )

        output_path.parent.mkdir(parents=True, exist_ok=True)
        image.save(output_path)

        return output_path

    def apply_consistency(
        self,
        generation_args: Dict,
        character_ids: List[str]
    ) -> Dict:
        """
        应用角色一致性到生成参数

        Args:
            generation_args: 原始生成参数
            character_ids: 需要保持的角色ID列表

        Returns:
            更新后的生成参数
        """
        if not self.config.use_ipadapter:
            return generation_args

        # 收集参考图像
        ref_images = []
        for char_id in character_ids:
            if char_id in self.reference_images:
                ref_images.append(self.reference_images[char_id])

        if not ref_images:
            return generation_args

        # 应用IPAdapter
        generation_args = self._apply_ipadapter(
            generation_args,
            ref_images
        )

        return generation_args

    def _apply_ipadapter(
        self,
        generation_args: Dict,
        reference_images: List[Path]
    ) -> Dict:
        """应用IPAdapter到pipeline"""
        # 加载IPAdapter (如果未加载)
        if self.ipadapter_pipeline is None:
            self._load_ipadapter()

        # 加载参考图像
        ref_imgs = [Image.open(p) for p in reference_images]

        # 应用IPAdapter参数
        generation_args['ip_adapter_image'] = ref_imgs[0] if len(ref_imgs) == 1 else ref_imgs
        generation_args['ip_adapter_scale'] = 0.6  # IPAdapter强度

        return generation_args

    def _build_character_prompt(self, char_info: Character) -> str:
        """构建角色prompt"""
        parts = [
            "masterpiece, best quality",
            f"1{char_info.gender}",  # 1boy 或 1girl
            f"{char_info.appearance}",
            f"{char_info.personality}",
            "portrait, looking at viewer, simple background"
        ]
        return ", ".join(parts)
```

## 缓存设计

### 图像生成缓存键

```python
# 图像缓存键格式
"img:{chapter_num}:{scene_num}:{hash(prompt)}:{seed}"

# 示例
"img:1:3:a1b2c3d4:12345"
```

### 缓存检查流程

```
开始生成图像
    │
    ▼
构建缓存键 ──▶ 检查磁盘缓存
    │              │
    │              存在 ──▶ 验证完整性
    │                            │
    │                          完整 ──▶ 返回缓存
    │                            │
    │                          不完整
    │                            │
    │                            ▼
    │                        重新生成
    │                            │
    │                            ▼
    │                        保存新缓存
    │                            │
    └────────────────────────────┘
```

## 命令行接口

```bash
# 完整图像生成
python run_stage2.py --novel-dir "outputs/novels/我的修仙小说"

# 只生成指定章节
python run_stage2.py --novel-dir "outputs/novels/我的修仙小说" --chapter 1

# 只生成指定场景
python run_stage2.py --novel-dir "outputs/novels/我的修仙小说" --scene ch1_sc1

# 强制重新生成（忽略缓存）
python run_stage2.py --novel-dir "outputs/novels/我的修仙小说" --force

# 使用不同模型
python run_stage2.py --novel-dir "outputs/novels/我的修仙小说" --model flux

# 指定分辨率
python run_stage2.py --novel-dir "outputs/novels/我的修仙小说" --width 1024 --height 1536
```
